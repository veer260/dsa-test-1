class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        // int nz = 0;
        int z = 0;
        int i = 0;

        while(i < nums.size()) {
            if(nums[i] != 0) {
                swap(nums[i], nums[z]);
                i++;
                z++;
            }else {
                i++;
            }
        }
        
        
    }
};
